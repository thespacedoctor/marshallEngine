98 # SELECT THE DATA REQUIRED FOR THIS LIGHTCURVE PLOT 

99 sqlQuery = """ 

100 SELECT 

101 transientBucketId, 

102 survey, 

103 magnitude, 

104 magnitudeError, 

105 observationDate, 

106 observationMJD, 

107 IFNULL(filter, '?') as filter, 

108 name, 

109 limitingMag, 

110 concat(ROUND(observationMJD, 2), SUBSTRING(filter, 1, 1)) as uniqueConstraint, 

111 concat(ROUND(observationMJD, 0), SUBSTRING(filter, 1, 1)) as limitConstraint 

112 FROM 

113 transientBucket 

114 WHERE 

115 replacedByRowId = 0 

116 AND transientBucketId = %(transientBucketId)s 

117 AND magnitude IS NOT NULL 

118 -- AND filter is not null

120 """ % locals() 

121 transientData = readquery( 

122 sqlQuery=sqlQuery, 

123 dbConn=self.dbConn, 

124 log=self.log 

125 ) 

126 

127 theseMags = [] 

128 theseMags[:] = [a 

129 for a in transientData if a["limitingMag"] == 0] 

130 

131 # CLIP OUTLIERS 

132 if len(theseMags) > 4: 

133 allmags = [] 

134 allmags[:] = [a["magnitude"] for a in theseMags] 

135 mean = np.mean(allmags) 

136 std = np.std(allmags) 

137 

138 if std > 0.1: 

139 allmags[:] = [a for a in transientData if (abs( 

140 a["magnitude"] - mean) < 2 * std) or a["limitingMag"] == 1] 

141 transientData = allmags 

142 

143 # NO DATA? 

144 if len(transientData) == 0: 

145 return False, False, False 

146 

147 # SPLIT DATA BY FILTER - MAGNITUDES AND LIMITS 

148 filterList = [] 

149 filterList[:] = set([r["filter"][0] for r in transientData]) 

150 

151 dataset = {} 

152 flatdata = {"mag": [], "mjd": []} 

153 flatLimits = {"mag": [], "mjd": []} 

154 for f in filterList: 

155 mag = [] 

156 magErr = [] 

157 magMjd = [] 

158 limit = [] 

159 limitMjd = [] 

160 magNoErr = [] 

161 magNoErrMjd = [] 

162 magNoErrFudge = [] 

163 catch = [] 

164 limitCatcher = {} 

165 for r in transientData: 

166 if r["filter"][0] == f and r["uniqueConstraint"] not in catch: 

167 if r["limitingMag"] == 0 and r["magnitudeError"]: 

168 mag.append(r["magnitude"]) 

169 magErr.append(r["magnitudeError"]) 

170 magMjd.append(r["observationMJD"]) 

171 flatdata["mag"].append(r["magnitude"]) 

172 flatdata["mjd"].append(r["observationMJD"]) 

173 catch.append(r["uniqueConstraint"]) 

174 elif r["limitingMag"] == 0 and not r["magnitudeError"]: 

175 magNoErr.append(r["magnitude"]) 

176 magNoErrFudge.append(0.3) 

177 magNoErrMjd.append(r["observationMJD"]) 

178 flatdata["mag"].append(r["magnitude"]) 

179 flatdata["mjd"].append(r["observationMJD"]) 

180 catch.append(r["uniqueConstraint"]) 

181 elif r["limitConstraint"] not in limitCatcher: 

182 limitCatcher[r["limitConstraint"]] = [ 

183 r["magnitude"], r["observationMJD"]] 

184 elif limitCatcher[r["limitConstraint"]][0] < r["magnitude"]: 

185 limitCatcher[r["limitConstraint"]] = [ 

186 r["magnitude"], r["observationMJD"]] 

187 

188 for k, v in list(limitCatcher.items()): 

189 limit.append(v[0]) 

190 limitMjd.append(v[1]) 

191 flatLimits["mag"].append(v[0]) 

192 flatLimits["mjd"].append(v[1]) 

193 

194 dataset[f] = { 

195 "limit": limit, 

196 "limitMjd": limitMjd, 

197 "mag": mag, 

198 "magErr": magErr, 

199 "magMjd": magMjd, 

200 "magNoErr": magNoErr, 

201 "magNoErrFudge": magNoErrFudge, 

202 "magNoErrMjd": magNoErrMjd 

203 } 

204 

205 if len(flatdata["mag"]) == 0: 

206 return False, False, False 

207 

208 self.log.debug('completed the ``_select_data_for_transient`` method') 

209 return dataset, flatdata, flatLimits
